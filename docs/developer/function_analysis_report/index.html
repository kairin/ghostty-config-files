Phase 1: Automated Static Analysis (Continuous)

1.1 ShellCheck Integration

Install ShellCheck (if not already installed):

# Ubuntu/Debian
sudo apt install shellcheck

# From source (latest version)
scversion="stable"
wget -qO- "https://github.com/koalaman/shellcheck/releases/download/${scversion}/shellcheck-${scversion}.linux.x86_64.tar.xz" | tar -xJv
sudo cp "shellcheck-${scversion}/shellcheck" /usr/bin/

Add to Pre-Commit Hook (.runners-local/workflows/pre-commit-local.sh):

# Add ShellCheck validation
echo "Running ShellCheck analysis..."
shellcheck_errors=0
while IFS= read -r script; do
    if ! shellcheck "$script"; then
        ((shellcheck_errors++))
    fi
done < <(fd -e sh . lib/ scripts/ tests/)

if [[ $shellcheck_errors -gt 0 ]]; then
    echo "ERROR: $shellcheck_errors scripts failed ShellCheck"
    exit 1
fi

Add to CI/CD Workflow (.runners-local/workflows/gh-workflow-local.sh):

validate_shellcheck() {
    log "Validating shell scripts with ShellCheck..."
    
    shellcheck --version
    
    # Run on all shell scripts
    fd -e sh . lib/ scripts/ tests/ -x shellcheck --severity=warning {}
    
    log "ShellCheck validation complete"
}

3.2 Documentation Linter

Create /home/kkk/Apps/ghostty-config-files/scripts/utils/lint_function_docs.sh:

#!/bin/bash
# Verify all functions have proper documentation

undocumented=()

while IFS=: read -r file line type func_name; do
    # Check if function has documentation comment above it
    doc_line=$((line - 1))
    
    if ! sed -n "${doc_line}p" "$file" | grep -q "^#"; then
        undocumented+=("${file}:${line}:${func_name}")
    fi
done < /tmp/function_definitions.txt

if [[ ${#undocumented[@]} -gt 0 ]]; then
    echo "WARNING: ${#undocumented[@]} functions lack documentation"
    printf '%s\n' "${undocumented[@]}" | head -10
fi
